create table CAR_SHARE_OFFICE
(
	USERNAME VARCHAR NOT NULL REFERENCES USER (USERNAME),
	SEEK_ID INTEGER NOT NULL REFERENCES SEEK (SEEK_ID),
	AGREEMENT_ID INTEGER NOT NULL,
	ADJUSTMENT_ID INTEGER NOT NULL,
	CONSULTANT_ID INTEGER NOT NULL,
	CORPORATE_ID INTEGER NOT NULL,
	PAYMENT_ID INTEGER NOT NULL,
	CURRENT_ACCOUNT_ID INTEGER NOT NULL,
	CONSTRAINT USERNAME_PK PRIMARY KEY (USERNAME, SEEK_ID)	
)
