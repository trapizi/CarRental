CREATE TABLE CORPORATE_MEMBER
(
	MEMBER_ID INT NOT NULL CONSTRAINT CORPORATE_MEMBER_FK1
		REFERENCES MEMBER ON DELETE CASCADE ON UPDATE RESTRICT,
	CORPORATE_ID INT NOT NULL CONSTRAINT CORPORATE_MEMBER_FK2
		REFERENCES CORPORATE ON DELETE CASCADE ON UPDATE RESTRICT,
	CONSTRAINT CORPORATE_MEMBER_PK
		PRIMARY KEY (MEMBER_ID, CORPORATE_ID)
)
