create table AGREEMENT
(
	AGREEMENT_ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	STATUS LONG VARCHAR,
	PAYAMT LONG VARCHAR,
	AGREEDATE LONG VARCHAR,
	CREATEDAY LONG VARCHAR,
	INITIATEBY LONG VARCHAR,
	OFFERRECEIVABLE DOUBLE,
	TOPIN BIGINT,
	FROMPIN BIGINT,
	PUPFROM BIGINT,
	PUPTO BIGINT
)
