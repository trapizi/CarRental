CREATE TABLE MEMBER
(
	MEMBER_ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	USERNAME VARCHAR(255),
	PASSWORD LONG VARCHAR,
	FIRST_NAME LONG VARCHAR,
	LAST_NAME LONG VARCHAR,
	EMAIL LONG VARCHAR,
	PHONE INTEGER,
	HOME_ADDRESS LONG VARCHAR,
	ACCOUNT_EXPIRY DATE,
	UNIQUE (USERNAME)
)
