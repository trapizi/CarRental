CREATE TABLE MEMBERSHIP_PAYMENT
(
	MEMBERSHIP_PAYMENT_ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	MEMBER_ID INTEGER,
	PAYMENT_AMOUNT FLOAT,
	PAYMENT_DATE DATE,
	PAYMENT_ACCOUNT LONG VARCHAR,
	PAYMENT_TYPE LONG VARCHAR,
	ACCOUNT_EXPIRY DATE,
	ACCOUNT_OWNER_NAME LONG VARCHAR,
	REFUND_FLAG BOOLEAN
)
