create table OFFER
(
	OFFER_ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	MEMBER_ID INTEGER NOT NULL CONSTRAINT MEMBER_ID_FK1
	REFERENCES MEMBER ON DELETE CASCADE ON UPDATE RESTRICT,
	SEATS INTEGER NOT NULL,
	CAR_TYPE LONG VARCHAR NOT NULL,
	BRAND LONG VARCHAR NOT NULL,
	MODEL LONG VARCHAR NOT NULL,
	TRANSMISSION LONG VARCHAR NOT NULL,
	FUEL_TYPE LONG VARCHAR NOT NULL,
	PRICE FLOAT NOT NULL,
	POSTCODE INTEGER NOT NULL,
	DRIVE_DAY DATE NOT NULL
)
