create table EMPLOYEE
(
	EMPLOYEE_ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	FIRST_NAME LONG VARCHAR,
	LAST_NAME LONG VARCHAR,
	EMAIL LONG VARCHAR,
	PHONE_NUMBER LONG VARCHAR,
	HIRE_DATE DATE,
	JOB_ID LONG VARCHAR,
	SALARY INTEGER,
	COMMISSION_PCT DOUBLE,
	MANAGER_ID INTEGER,
	DEPARTMENT_ID INTEGER
)
